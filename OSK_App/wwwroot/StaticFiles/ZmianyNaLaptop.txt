DROP TABLE PAYMENTS;

CREATE TABLE TYPEPAYMENTS(
	ID INT PRIMARY KEY IDENTITY(1,1),
	Name VARCHAR(50)
);

-- Tabela Wplata
CREATE TABLE PAYMENTS(
	ID INT PRIMARY KEY IDENTITY(1,1),
	Cost DECIMAL(5,2),
	PaymentDate VARCHAR(10),
	PaymentTime VARCHAR(5),
	ReceiptDate VARCHAR(10),
	StudentID INT, CONSTRAINT FK_Payment_Student FOREIGN KEY(StudentID) REFERENCES STUDENTS(UserID),
	EmployeeID INT, CONSTRAINT FK_Payment_Employee FOREIGN KEY(EmployeeID) REFERENCES EMPLOYEES(UserID),
	PaymentStatID INT, CONSTRAINT FK_Payment_PaymentStatus FOREIGN KEY(PaymentStatID) REFERENCES PAYMENTSTATUSES(ID),
	TypePaymentID INT, CONSTRAINT FK_Payment_TypePayment FOREIGN KEY(TypePaymentID) REFERENCES TYPEPAYMENTS(ID)
);

-- Tabela Kurs kursanta
CREATE TABLE STUDENTCOURSES(
	ID INT PRIMARY KEY IDENTITY(1,1),
	CourseID INT, CONSTRAINT FK_StudentCourse_Course FOREIGN KEY(CourseID) REFERENCES COURSES(ID),
	StartDate VARCHAR(10),
	EndDate VARCHAR(10),
	StudentID INT, CONSTRAINT FK_StudentCourse_Student FOREIGN KEY(StudentID) REFERENCES STUDENTS(UserID),
	ExtraHours INT DEFAULT 0,
	SumOfPayment INT DEFAULT 0,
	StudentCourseStatID INT, CONSTRAINT FK_StudentCourse_StudentCourseStatus FOREIGN KEY(StudentCourseStatID) REFERENCES STUDENTCOURSESTATUTES(ID)
);
